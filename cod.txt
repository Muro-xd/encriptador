let numSecreto = 0;
let intentos = 1;
let listaNumeroSorteado = [];
let numMax = 10;

function buttomTry(){
    let numUser = parseInt(document.getElementById("valor").value);
    if(numUser===numSecreto){
        asignarTxtElement('p',`Acertaste el numero secreto en ${intentos} ${intentos==1?'vez':'veces'}`);
        document.getElementById('reiniciar').removeAttribute('disabled');
    }else{
        
        if(numUser<numSecreto){
            asignarTxtElement('p',"El numero es mayor");
        }else{
            asignarTxtElement('p',"El numero es menor")
        }
    intentos++;
    }
    limpiar();
}
function asignarTxtElement(elemento,texto){
    let etiqueta = document.querySelector(elemento);
    etiqueta.innerHTML = texto;
}
function numRandom(){
    let numGenerado = Math.floor(Math.random()*numMax)+1;
    if(listaNumeroSorteado.length == numMax){
        asignarTxtElement('p','Ya se sortearon todos los numeros posibles');
    }else{
        if(listaNumeroSorteado.includes(numGenerado)){
            return numRandom();
        }else{
            listaNumeroSorteado.push(numGenerado);
            return numGenerado;
        }
    }
}
function limpiar(){
    document.getElementById("valor").value='';
    //document.querySelector('#valor').value='';
}
function valoresInicial(){
    asignarTxtElement('h1',"Juego del numero secreto");
    asignarTxtElement('p',`Indica un numero del 1 al ${numMax}`);
    numSecreto = numRandom();
    intentos = 1;
}
function restartGame(){
    //limpiar caja
    limpiar();
    //indicar mensaje de intervalos
    //generar numero aleatorio
    //inicializar el numero de intentos
    valoresInicial();
    //desabilitar el boton de nuevo juego
    document.getElementById("reiniciar").setAttribute('disabled','true');
}
valoresInicial();
